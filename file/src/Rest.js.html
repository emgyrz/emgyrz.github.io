    // @flow

import { isObj, isStr } from './is'
import { parseObj as parseJson } from './json'
import { hasProp } from './obj'
import prepareBody, { type AsType } from './prepareRequestBody'
import type {
  StringsMap, AnyMap, HttpMethods, RespType
} from './_types'

type InputOptsType = {|
  baseUrl: string,
  defaultHeaders?: StringsMap
|}

type AddonRequestOptsType = {|
  headers?: StringsMap,
  as?: AsType,
  query?: AnyMap
|}


/**
 * Отправка запросов
 *
 * @export
 * @class Rest
 */
export default class Rest {
  /**
   * Адрес api без роутов
   *
   * @type {string}
   * @example
   * https://api.addreality.com
   * @memberof Rest
   */
  baseUrl: string

  /**
   * Заголовки, отправляемые с каждым запросом
   *
   * @type {StringsMap}
   * @example
   * {
   *   'X-Access-Token': 'qwfg34gtrb
   * }
   * @memberof Rest
   */
  defaultHeaders: StringsMap


  /**
   *Creates an instance of Rest.
   * @param {{ baseUrl: string, defaultHeaders?: StringsMap }} opts
   * @memberof Rest
   */
  constructor( opts: InputOptsType ) {
    this.baseUrl = opts.baseUrl.replace( /\/$/, '' )
    this.defaultHeaders = isObj( opts.defaultHeaders ) ? opts.defaultHeaders : {}
  }

  /**
   * Возвращает полный url
   *
   * @param {string} url
   * @param {mixed} [query]
   * @example
   * this.getUrl( '/devices', '?status=1' )
   * // https://api.addreality.com/devices?status=1
   * @returns {string}
   * @memberof Rest
   */
  getUrl( url: string, query?: mixed ): string {
    return this.baseUrl + url + ( isStr( query ) ? query : '' )
  }

  _getHeaders( headers?: StringsMap ): StringsMap {
    return Object.assign( {}, this.defaultHeaders, isObj( headers ) ? headers : {} )
  }

  // eslint-disable-next-line max-params
  _prepareAndSend( method: HttpMethods, urlIn: string, bodyIn?: ?AnyMap, optsIn?: AddonRequestOptsType ) {
    const opts = isObj( optsIn ) ? optsIn : {}
    const query = prepareBody( { body: opts.query, as: 'query' } )
    const url = this.getUrl( urlIn, query )
    return window.fetch( url, {
      method,
      headers: this._getHeaders( opts.headers ),
      body: prepareBody( { body: bodyIn, as: opts.as } )
    } )
  }



  static _handleResp( promise: Promise<Response> ): Promise<RespType> {
    const unknowErr = { code: -1, message: 'unknown error' }

    let sts

    return promise.then( ( resp: Response ) => {
      sts = resp.status
      return resp.text()
    } )
      .then( ( respText: string ) => {
        const parsed = respText === '' ? {} : parseJson( respText )
        const isParsedNull = parsed === null
        const data = parsed === null ? {} : parsed

        // const isSplitterResp = hasProp( data, 'response' ) || hasProp( data, 'error' )

        const hasErr = hasProp( data, 'error' ) || sts !== 200

        if ( hasErr ) {
          let errData: AnyMap = isParsedNull ? unknowErr : data
          if ( hasProp( data, 'error' ) ) {
            // $FlowFixMe
            errData = data.error
          }
          return {
            hasErr,
            status: sts,
            respText,
            error: errData
          }
        }
        return {
          hasErr,
          status: sts,
          respText,
          // $FlowFixMe
          data: hasProp( data, 'response' ) ? data.response : data
        }
      } )
      .catch( ( error: AnyMap ) => ( {
        hasErr: true,
        status: sts,
        respText: '',
        error
      } ) )
  }


  /**
   * Метод GET-запроса
   *
   * @param {string} urlIn
   * @param {?AnyMap} bodyIn
   * @param {{ headers?: StringsMap, as?: AsType, query?: AnyMap }} [optsIn]
   * @returns
   * @memberof Rest
   */
  get( urlIn: string, bodyIn: ?AnyMap, optsIn?: AddonRequestOptsType ) {
    const opts = isObj( optsIn ) ? optsIn : {}
    const url = this.getUrl( urlIn, prepareBody( { body: bodyIn, as: 'query' } ) )
    return Rest._handleResp( window.fetch( url, {
      method: 'GET',
      headers: this._getHeaders( opts.headers ),
      body: null
    } ) )
  }

  /**
   * Метод POST-запроса
   *
   * @param {string} url
   * @param {?AnyMap} body
   * @param {{ headers?: StringsMap, as?: AsType, query?: AnyMap }} [opts]
   * @returns
   * @memberof Rest
   */
  post( url: string, body: ?AnyMap, opts?: AddonRequestOptsType ) {
    return Rest._handleResp( this._prepareAndSend( 'POST', url, body, opts ) )
  }


  /**
   * Метод PUT-запроса
   *
   * @param {string} url
   * @param {?AnyMap} body
   * @param {{ headers?: StringsMap, as?: AsType, query?: AnyMap }} [opts]
   * @returns
   * @memberof Rest
   */
  put( url: string, body: ?AnyMap, opts?: AddonRequestOptsType ) {
    return Rest._handleResp( this._prepareAndSend( 'PUT', url, body, opts ) )
  }



  /**
   * Метод DELETE-запроса
   *
   * @param {string} url
   * @param {?AnyMap} body
   * @param {?{ headers?: StringsMap, as?: AsType, query?: AnyMap }} [opts]
   * @returns
   * @memberof Rest
   */
  del( url: string, body: ?AnyMap, opts?: AddonRequestOptsType ) {
    return Rest._handleResp( this._prepareAndSend( 'DELETE', url, body, opts ) )
  }

}
