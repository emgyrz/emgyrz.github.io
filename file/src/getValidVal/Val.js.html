import { isStr, isF, isObj } from '../is'
import type {
  ObjPropsValsType, Types, defaultVal, CollectedValProps
} from './_types'


class Val {
  name: string = 'unknown val';
  _type: Types;
  _isRequired: boolean = false;
  _fillIfNot: boolean = true;
  _defaultVal: ?defaultVal;
  _arrItem: ?Val;
  _props: ?ObjPropsValsType;

  constructor( name?: string ) {
    if ( isStr( name ) ) {
      this.name = name
    }
  }

  static define( name?: string ): Val {
    return new Val( name )
  }

  static get availableTypes(): Types[] {
    return [ 'string', 'number', 'boolean', 'object', 'array', 'null' ]
  }

  static err( msg: string ) {
    throw new Error( msg )
  }

  get _isTypeSetted(): boolean {
    return isStr( this._type )
  }

  _checkTypeSetting() {
    if ( this._isTypeSetted ) {
      Val.err( `${this.name} is allready have a type` )
    }
  }

  default( val: defaultVal ): Val {
    this._defaultVal = val
    return this
  }

  hasItems( val: Val ): Val {
    if ( this._type !== 'array' ) {
      throw new Error( `${this.name} is not an array to contains array items` )
    }
    this._arrItem = val
    return this
  }
  hasProperties( props: ObjPropsValsType ): Val {
    if ( this._type !== 'object' ) {
      throw new Error( `${this.name} is not an object to have properties` )
    }
    this._props = props
    return this
  }

  get required(): Val {
    this._isRequired = true
    return this
  }

  get string(): Val {
    this._checkTypeSetting()
    this._type = 'string'
    return this
  }
  get number(): Val {
    this._checkTypeSetting()
    this._type = 'number'
    return this
  }
  get boolean(): Val {
    this._checkTypeSetting()
    this._type = 'boolean'
    return this
  }
  get object(): Val {
    this._checkTypeSetting()
    this._type = 'object'
    return this
  }
  get array(): Val {
    this._checkTypeSetting()
    this._type = 'array'
    return this
  }
  // TODO: move
  get null(): Val {
    this._checkTypeSetting()
    this._type = 'null'
    return this
  }

  get dontFill(): Val {
    this._fillIfNot = false
    return this
  }

  _collect(): CollectedValProps {
    const { name } = this
    const type = this._type

    if ( !isStr( type ) || !Val.availableTypes.includes( type ) ) {
      Val.err( `type is not setted or wrong for val ${name}` )
    }

    // debugger // eslint-disable-line no-debugger

    const isRequired = this._isRequired
    const hasDefault = Object.prototype.hasOwnProperty.call( this, '_defaultVal' )
    && this._defaultVal !== undefined
    const fillIfNot = this._fillIfNot === true

    // if ( hasDefault && fillIfNot ) {
    //   Val.err( `you dont't need to set together default value and cast value for ${name}` )
    // }

    const result: CollectedValProps = {
      name,
      type,
      isRequired,
      hasDefault,
      fillIfNot
    }

    if ( hasDefault ) {
      const def = this._defaultVal
      result.defaultVal = isF( def ) ? def() : def
    }

    if ( type === 'object' ) {
      if ( !isObj( this._props ) ) {
        Val.err( `${name} is object but for it not defined properties` )
      } else {
        const pr: ObjPropsValsType = this._props
        Object.keys( pr ).forEach( ( propKey: string ) => {
          if ( !( pr[ propKey ] instanceof Val ) ) {
            Val.err( `property ${propKey} for object ${name} is wrong defined` )
          }
        } )
        result.props = pr
      }
    }

    if ( type === 'array' ) {
      if ( !( this._arrItem instanceof Val ) ) {
        Val.err( `${name} is array but it items type is not defined` )
      } else {
        result.arrItem = this._arrItem
      }
    }

    return result
  }

  //
  // Short creating Val aliases
  //
  static str( name?: string ): Val {
    return Val.define( name ).string
  }
  static num( name?: string ): Val {
    return Val.define( name ).number
  }
  static bool( name?: string ): Val {
    return Val.define( name ).boolean
  }
  static arr( name?: string, arrItem?: ?Val ): Val {
    const arrVal = Val.define( name ).array
    if ( arrItem instanceof Val ) {
      return arrVal.hasItems( arrItem )
    }
    return arrVal
  }
  static obj( name?: string, props?: ?ObjPropsValsType ): Val {
    const objVal = Val.define( name ).object
    if ( isObj( props ) ) {
      return objVal.hasProperties( props )
    }
    return objVal
  }

}


export default Val
