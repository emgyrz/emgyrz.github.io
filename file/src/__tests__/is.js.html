    import * as is from '../is'


const rightValsFor = {
  isArr: [],
  isF: () => {},
  isNum: 0,
  isNull: null,
  isUndef: undefined,
  isNil: undefined,
  isStr: '',
  isBool: true,
  isObj: { x: 3 },
  isBlob: new Blob( [ 1, 2, 3 ] ),
  isFile: new File( [ 1, 2, 3 ], 'file' )
}

const allVals = Object.values( rightValsFor )

describe( 'is', () => {
  test( 'все функции is... кроме isNil корректно проверяют входящие значения ТОЛЬКО своего типа', () => {
    Object.keys( is ).forEach( funcName => {
      if (
        funcName === 'isEl'
        || funcName === 'isSVG'
        || funcName === 'isNullSome'
      ) return

      const checker = is[ funcName ]
      const rightVal = rightValsFor[ funcName ]
      if ( !rightValsFor.hasOwnProperty( funcName ) ) {
        throw new Error( `для функции ${funcName} не заданы правильные параметры` )
      }
      expect( checker( rightVal ) ).toBe( true )
      if ( funcName !== 'isNil' ) {
        expect( allVals.every( val => val === rightVal || checker( val ) === false ) ).toBe( true )
      }
    } )
  } )

  test( 'isObj ожидает только простой объект', () => {
    expect( is.isObj( [] ) ).toBe( false )
    expect( is.isObj( () => {} ) ).toBe( false )
  } )

  test( 'isNil ожидает и null, и undefined', () => {
    expect( is.isNil( null ) ).toBe( true )
    expect( is.isNil( undefined ) ).toBe( true )
  } )

} )
